```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define N 5
#define N1 397
#define N2 476
#define N3 21

int main() {
    int cnt;
    int random_major, random_no;

    srand(time(NULL));      // 以当前系统时间作为随机种子

    cnt = 0;
    while(cnt < N) {
        random_major = rand() % 2;

        if(random_major) {
            random_no = rand() % (N2 - N1 + 1) + N1;
            printf("20248329%04d\n", random_no);
        }
        else {
            random_no = rand() % N3 + 1;
            printf("20248395%04d\n", random_no);
        }

        cnt++;
    }

    return 0;
}
```

# 任务二

## 验证性实验 根据一元二次方程求解公式，编程求解一元二次方程

### 源码

```c
// 一元二次方程求解

#include <stdio.h>
#include <math.h>

int main() {
    double a, b, c;
    double delta, p1, p2; // 用于保存中间计算结果

    while(scanf("%lf%lf%lf", &a, &b, &c) != EOF) {
        if(a == 0) {
            printf("a = 0, invalid input\n");
            continue;
        }

        delta = b*b - 4*a*c;
        p1 = -b/2/a;
        p2 = sqrt(fabs(delta))/2/a;

        if(delta == 0)
            printf("x1 = x2 = %.2g\n", p1);
        else if(delta > 0)
            printf("x1 = %.2g, x2 = %.2g\n", p1+p2, p1-p2);
        else {
            printf("x1 = %.2g + %.2gi, ", p1, p2);
            printf("x2 = %.2g - %.2gi\n", p1, p2);
        }
    }

    return 0;
}
```

# 任务四

## 编写一个模拟记账程序统计一天的开销

### 源码

```c
#include <stdio.h>

int main()
{
    float input;
    float sum = 0, max = 0, min = 20000;
    printf("Enter expense (-1 to stop):\n");
    for (int i = 0; scanf("%f", &input) && input != -1; i++)
    {
        sum += input;
        if (input > max)
            max = input;
        if (input < min)
            min = input;
    }

    printf("Today's total expense:%.1f\nToday's biggest expense:%.1f\nToday's smallest expense:%.1f\n", sum, max, min);

    return 0;
}
```

## 编写程序 根据输入的三角形三边边长，判断三角形类型

### 源码

```c
#include <stdio.h>

int main()
{
    int a, b, c;
    while (scanf("%d%d%d", &a, &b, &c) != EOF)
    {
        if (a + b > c && a + c > b && b + c > a)
        {
            if (a == b || a == c || b == c)
            {
                if (a == b && b == c)
                {
                    printf("等边三角形\n");
                }
                else
                {
                    printf("等腰三角形\n");
                }
            }
            else if (a * a + b * b == c * c || a * a + c * c == b * b || c * c + b * b == a * a)
            {
                printf("直角三角形\n");
            }
            else
            {
                printf("普通三角形\n");
            }
        }
        else
        {
            printf("不能构成三角形\n");
        }
    }
    return 0;
}
```
